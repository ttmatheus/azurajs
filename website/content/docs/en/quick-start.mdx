## Step 1: Configuration File âš™ï¸

The configuration file controls server behavior. Create it in the project root:

<Tabs items={['TypeScript', 'JavaScript']}>
  <Tab value="TypeScript">
    ```typescript title="azura.config.ts"
    import type { ConfigTypes } from "azurajs/config";

    const config: ConfigTypes = {
      // Development environment
      environment: "development",
      
      // Server settings
      server: {
        port: 3000,              // Port where the server will run
        cluster: false,          // Disable cluster in dev
        ipHost: true,            // Auto-detect IP
        https: false,            // Simple HTTP (for dev)
      },
      
      // Logging settings
      logging: {
        enabled: true,           // Enable logs
        showDetails: true,       // Show request details
      },
      
      // Optional plugins
      plugins: {
        // CORS - allow requests from other domains
        cors: {
          enabled: true,
          origins: ["*"],        // Accept any origin (dev only!)
          methods: ["GET", "POST", "PUT", "DELETE", "PATCH"],
          allowedHeaders: ["Content-Type", "Authorization"],
        },
        
        // Rate Limiting - protection against abuse
        rateLimit: {
          enabled: false,        // Disabled in dev
          limit: 100,            // Maximum 100 requests
          timeframe: 60000,      // Per minute (60000ms)
        },
      },
    };

    export default config;
    ```
  </Tab>
  
  <Tab value="JavaScript">

## Step 2: Create the Controller ğŸ¯

Now let's create a controller that manages users. AzuraJS supports two approaches:

<Tabs items={['TypeScript (Decorators)', 'JavaScript (Functional)']}>
  <Tab value="TypeScript (Decorators)">
    Create `src/controllers/UserController.ts`:

    ```typescript title="src/controllers/UserController.ts"
    import { 
      Controller, 
      Get, 
      Post, 
      Put, 
      Delete, 
      Body, 
      Param, 
      Res 
    } from "azurajs/decorators";
    import type { ResponseServer } from "azurajs/types";

    // In-memory database (replace with real DB in production)
    const users = [
      { id: 1, name: "Alice", email: "alice@example.com" },
      { id: 2, name: "Bob", email: "bob@example.com" },
    ];

    @Controller("/api/users")  // Define the base route prefix
    export class UserController {
      
      // GET /api/users - List all users
      @Get()
      getAllUsers(@Res() res: ResponseServer) {
        res.json({ 
          success: true, 
          data: users,
          total: users.length
        });
      }

      // GET /api/users/:id - Get specific user
      @Get("/:id")
      getUserById(
        @Param("id") id: string,  // Capture URL parameter
        @Res() res: ResponseServer
      ) {
        const user = users.find(u => u.id === Number(id));
        
        if (!user) {
          return res.status(404).json({ 
            success: false, 
            error: "User not found" 
          });
        }
        
        res.json({ success: true, data: user });
      }

      // POST /api/users - Create new user
      @Post()
      createUser(
        @Body() body: any,  // Capture request body
        @Res() res: ResponseServer
      ) {
        // Basic validation
        if (!body.name || !body.email) {
          return res.status(400).json({
            success: false,
            error: "Name and email are required"
          });
        }

        const newUser = {
          id: users.length + 1,
          name: body.name,
          email: body.email,
        };
        
        users.push(newUser);
        
        res.status(201).json({ 
          success: true, 
          data: newUser,
          message: "User created successfully"
        });
      }

      // PUT /api/users/:id - Update existing user
      @Put("/:id")
      updateUser(
        @Param("id") id: string,
        @Body() body: any,
        @Res() res: ResponseServer
      ) {
        const user = users.find(u => u.id === Number(id));
        
        if (!user) {
          return res.status(404).json({ 
            success: false, 
            error: "User not found" 
          });
        }
        
        // Update only provided fields
        if (body.name) user.name = body.name;
        if (body.email) user.email = body.email;
        
        res.json({ 
          success: true, 
          data: user,
          message: "User updated successfully"
        });
      }

      // DELETE /api/users/:id - Remove user
      @Delete("/:id")
      deleteUser(
        @Param("id") id: string,
        @Res() res: ResponseServer
      ) {
        const index = users.findIndex(u => u.id === Number(id));
        
        if (index === -1) {
          return res.status(404).json({ 
            success: false, 
            error: "User not found" 
          });
        }
        
        users.splice(index, 1);
        
        res.json({ 
          success: true, 
          message: "User deleted successfully" 
        });
      }
    }
    ```

    <Callout type="tip" title="ğŸ’¡ Tip">
    **Decorators** allow you to organize routes elegantly and type-safely. Each controller method corresponds to an HTTP route!
    </Callout>
  </Tab>

  <Tab value="JavaScript (Functional)">
    For those who prefer JavaScript or a more functional approach, create `src/index.js`:

    ```javascript title="src/index.js"
    const { AzuraClient } = require("azurajs");

    // In-memory database (replace with real DB in production)
    const users = [
      { id: 1, name: "Alice", email: "alice@example.com" },
      { id: 2, name: "Bob", email: "bob@example.com" },
    ];

    // Create the server instance
    const app = new AzuraClient();

    // GET /api/users - List all users
    app.get("/api/users", ({ req, res }) => {
      res.json({ 
        success: true, 
        data: users,
        total: users.length
      });
    });

    // GET /api/users/:id - Get specific user
    app.get("/api/users/:id", ({ req, res }) => {
      const { id } = req.params;  // Capture URL parameter
      const user = users.find(u => u.id === Number(id));
      
      if (!user) {
        return res.status(404).json({ 
          success: false, 
          error: "User not found" 
        });
      }
      
      res.json({ success: true, data: user });
    });

    // POST /api/users - Create new user
    app.post("/api/users", ({ req, res }) => {
      const body = req.body;  // Capture request body
      
      // Basic validation
      if (!body.name || !body.email) {
        return res.status(400).json({
          success: false,
          error: "Name and email are required"
        });
      }

      const newUser = {
        id: users.length + 1,
        name: body.name,
        email: body.email,
      };
      
      users.push(newUser);
      
      res.status(201).json({ 
        success: true, 
        data: newUser,
        message: "User created successfully"
      });
    });

    // PUT /api/users/:id - Update existing user
    app.put("/api/users/:id", ({ req, res }) => {
      const { id } = req.params;
      const body = req.body;
      const user = users.find(u => u.id === Number(id));
      
      if (!user) {
        return res.status(404).json({ 
          success: false, 
          error: "User not found" 
        });
      }
      
      // Update only provided fields
      if (body.name) user.name = body.name;
      if (body.email) user.email = body.email;
      
      res.json({ 
        success: true, 
        data: user,
        message: "User updated successfully"
      });
    });

    // DELETE /api/users/:id - Remove user
    app.delete("/api/users/:id", ({ req, res }) => {
      const { id } = req.params;
      const index = users.findIndex(u => u.id === Number(id));
      
      if (index === -1) {
        return res.status(404).json({ 
          success: false, 
          error: "User not found" 
        });
      }
      
      users.splice(index, 1);
      
      res.json({ 
        success: true, 
        message: "User deleted successfully"
      });
    });

    // Start the server
    app.listen();
    ```

    <Callout type="tip" title="ğŸ’¡ Tip">
    The **functional approach** is more familiar if you come from Express. It's straightforward, simple, and doesn't require decorators!
    </Callout>
  </Tab>
</Tabs>

## Step 3: Create Main File ğŸ–¥ï¸

<Tabs items={['TypeScript (Decorators)', 'JavaScript (Skip)']}>
  <Tab value="TypeScript (Decorators)">
    If you used decorators, you need to create a main file to register the controllers.

    Create `src/index.ts`:

    ```typescript title="src/index.ts"
    import { AzuraClient } from "azurajs";
    import { applyDecorators } from "azurajs/decorators";
    import { UserController } from "./controllers/UserController";

    // Initialize the server
    const app = new AzuraClient();

    // Register controllers
    // You can pass multiple controllers here
    applyDecorators(app, [
      UserController,
      // Add more controllers as needed
    ]);

    // Start the server
    app.listen().then(() => {
      console.log("âœ… API ready to use!");
    });
    ```

    <Callout type="info">
    The `applyDecorators()` function processes all controller decorators and registers routes automatically.
    </Callout>
  </Tab>

### ğŸŒ Environment Variables

Use environment variables for sensitive configuration:

```typescript title="azura.config.ts"
import type { ConfigTypes } from "azurajs/config";

const config: ConfigTypes = {
  environment: process.env.NODE_ENV || "development",
  server: {
    port: Number(process.env.PORT) || 3000,
    cluster: process.env.NODE_ENV === "production",
  },
  plugins: {
    cors: {
      enabled: true,
      origins: process.env.ALLOWED_ORIGINS?.split(",") || ["*"],
    },
    rateLimit: {
      enabled: process.env.NODE_ENV === "production",
      limit: Number(process.env.RATE_LIMIT) || 100,
    },
  },
};

export default config;
```
